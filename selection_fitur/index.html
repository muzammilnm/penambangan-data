            arr1.append(data[i])
    arr2 = []
    for t in range(len(arr1)):
        arr2.append(0)
    for u in range(len(data)):
        if(data[u] in arr1):
            arr2[arr1.index(data[u])] += 1
    rumus = 0
    for h in arr2:
        var = h/len(data)*math.log(h/len(data),2)
        rumus = rumus - var
    return rumus,len(arr2)
